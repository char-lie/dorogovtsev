\mdfsetup{skipbelow=\topskip}

\mdfdefinestyle{theoremstyle}{%
    apptotikzsetting={%
        \tikzset{mdfbackground/.append style ={%
                %top color=white!95!black,
                %bottom color=white
            },
            mdfframetitlebackground/.append style={
                top color=white!90!black,
                bottom color=white
            }
        }%
    },
    roundcorner=10pt,
    linewidth=3pt,
    middlelinewidth=1pt,
    shadow=true,
    %frametitlerule=true,
    %frametitlerulewidth=1pt,
    innertopmargin=10pt,%
    %nobreak=true
}%{theorem}{Теорема}[section]

\mdfdefinestyle{remarkstyle}{%
    rightline=false,
    %bottomline=false,
    %roundcorner=10pt,
    linecolor=gray,
    innertopmargin=5pt,
    frametitlerule=true,
    %frametitlerulewidth=1pt,
    %frametitlerulecolor=gray,
    %nobreak=true
}

\mdfdefinestyle{examplestyle}{%
    apptotikzsetting={%
        \tikzset{
                mdfframetitlebackground/.append style={
                right color=white!90!black,
                left color=white
            }
        }
    },
    leftline=false,
    topline=false,
    linecolor=white!90!black,
    linewidth=2pt,
    innertopmargin=3pt,
    %nobreak=true
}
